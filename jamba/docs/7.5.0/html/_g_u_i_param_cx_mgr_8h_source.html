216 inline TParam __registerListener(TParam iParam, Parameters::IChangeListener *iChangeListener)
217 {
218 if(iParam.exists() && iChangeListener)
219 fParamCxs.emplace_back(iParam.connect(iChangeListener));
220 return iParam;
221 }
