Detailed Description

template<typename TView>
class pongasoft::VST::GUI::Views::TCustomViewCreator< TView >

Generic custom view creator base class. Inherit from it and call the various "registerXX" methods in the constructor.

In case of inheritance, you do the following:

class CustomView1 : public CControl { ... } class CustomView2 : public CustomView1 { ... }

class CustomView1Creator : public CustomViewCreator<CustomView1> { public: explicit CustomView1Creator(char const *iViewName = nullptr, char const *iDisplayName = nullptr) : CustomViewCreator(iViewName, iDisplayName) { ... register CustomView1 attributes here ... } } class CustomView2Creator : public CustomViewCreator<CustomView2> { public: explicit CustomView2Creator(char const *iViewName = nullptr, char const *iDisplayName = nullptr) : CustomViewCreator(iViewName, iDisplayName) { registerAttributes(CustomView1Creator()); ... register CustomView2 attributes here ... } }
